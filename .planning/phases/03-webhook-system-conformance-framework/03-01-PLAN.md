    **src/webhook-delivery.ts:** Extract and generalize from twins/shopify/src/services/webhook-sender.ts:
    ```typescript
    import crypto from 'node:crypto';

    export function generateHmacSignature(payload: string, secret: string): string {
      return crypto.createHmac('sha256', secret).update(payload, 'utf8').digest('base64');
    }

    export async function deliverWebhook(delivery: WebhookDelivery, timeoutMs: number = 5000): Promise<void> {
      const body = JSON.stringify(delivery.payload);
      const signature = generateHmacSignature(body, delivery.secret);

      const headers: Record<string, string> = {
        'Content-Type': 'application/json',
        'X-Shopify-Hmac-Sha256': signature,
        'X-Shopify-Topic': delivery.topic,
        'X-Shopify-Webhook-Id': delivery.id,
        ...delivery.headers,
      };

      const response = await fetch(delivery.callbackUrl, {
        method: 'POST',
        headers,
        body,
        signal: AbortSignal.timeout(timeoutMs),
      });

      if (!response.ok) {
        throw new Error(`Webhook delivery failed: ${response.status} ${response.statusText}`);
      }
    }
    ```
    Note: Non-2xx responses MUST throw so the queue retries. The current webhook-sender swallows errors -- this is the upgrade.

    **src/webhook-queue.ts:** In-memory queue with configurable retry:
    ```typescript
    import { randomUUID } from 'node:crypto';
    import { deliverWebhook } from './webhook-delivery.js';
    import type { WebhookDelivery, WebhookJob, WebhookQueueOptions, DeadLetterStore } from './types.js';

    export class WebhookQueue {
      private pending: Map<string, { job: WebhookJob; timer: ReturnType<typeof setTimeout> | null }> = new Map();
      private timeScale: number;
      private retryDelays: number[];
      private deadLetterStore: DeadLetterStore;
      private syncMode: boolean;
      private deliveryTimeoutMs: number;
      private logger: WebhookQueueOptions['logger'];

      constructor(options: WebhookQueueOptions) {
        this.timeScale = options.timeScale ?? 1.0;
        this.retryDelays = options.retryDelays ?? [0, 60_000, 300_000];
        this.deadLetterStore = options.deadLetterStore;
        this.syncMode = options.syncMode ?? false;
        this.deliveryTimeoutMs = options.deliveryTimeoutMs ?? 5000;
        this.logger = options.logger;
      }

      async enqueue(delivery: WebhookDelivery): Promise<string> {
        const jobId = delivery.id || randomUUID();
        const now = Date.now();
        const job: WebhookJob = {
          delivery,
          jobId,
          attempt: 0,
          firstAttemptedAt: now,
          lastAttemptedAt: now,
        };

        if (this.syncMode) {
          await this.executeAttempt(job);
          return jobId;
        }

        this.pending.set(jobId, { job, timer: null });
        this.scheduleAttempt(jobId);
        return jobId;
      }

      /** Get count of in-flight deliveries */
      get size(): number { return this.pending.size; }

      /** Drain all pending jobs -- cancel timers, move remaining to DLQ */
      async drain(): Promise<void> {
        for (const [jobId, entry] of this.pending) {
          if (entry.timer) clearTimeout(entry.timer);
          // Optionally move to DLQ or just discard
        }
        this.pending.clear();
      }

      /** Shutdown cleanly -- cancel all pending timers */
      shutdown(): void {
        for (const [, entry] of this.pending) {
          if (entry.timer) clearTimeout(entry.timer);
        }
        this.pending.clear();
      }

      private scheduleAttempt(jobId: string): void {
        const entry = this.pending.get(jobId);
        if (!entry) return;

        const delay = entry.job.attempt === 0
          ? 0  // Immediate first attempt
          : (this.retryDelays[entry.job.attempt - 1] ?? this.retryDelays[this.retryDelays.length - 1]) * this.timeScale;

        if (delay === 0) {
          this.executeAttempt(entry.job).catch(() => {});
        } else {
          entry.timer = setTimeout(() => {
            this.executeAttempt(entry.job).catch(() => {});
          }, delay);
        }
      }

      private async executeAttempt(job: WebhookJob): Promise<void> {
        job.lastAttemptedAt = Date.now();
        try {
          await deliverWebhook(job.delivery, this.deliveryTimeoutMs);
          this.pending.delete(job.jobId);
          this.logger?.info({ jobId: job.jobId, topic: job.delivery.topic, attempt: job.attempt }, 'Webhook delivered');
        } catch (error) {
          const errorMsg = error instanceof Error ? error.message : String(error);
          job.lastError = errorMsg;
          job.attempt++;

          if (job.attempt > this.retryDelays.length) {
            // Max retries exhausted
            this.deadLetterStore.add(job);
            this.pending.delete(job.jobId);
            this.logger?.warn({ jobId: job.jobId, topic: job.delivery.topic, attempts: job.attempt, error: errorMsg }, 'Webhook moved to dead letter queue');
          } else {
            this.logger?.info({ jobId: job.jobId, topic: job.delivery.topic, attempt: job.attempt, nextRetryIn: this.retryDelays[job.attempt - 1] * this.timeScale }, 'Webhook delivery failed, scheduling retry');
            this.scheduleAttempt(job.jobId);
          }
        }
      }
    }
    ```

    **Important:** The retry delays array `[0, 60000, 300000]` means: attempt 0 = immediate, retry 1 after 1min, retry 2 after 5min. Total 3 attempts. After 3 failures, move to DLQ.
