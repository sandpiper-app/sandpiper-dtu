    ```typescript
    function parseLineItems(lineItemsJson: string | null): Record<number, number> {
      if (!lineItemsJson) return {};
      try {
        const items = JSON.parse(lineItemsJson);
        const map: Record<number, number> = {};
        for (const item of items) {
          map[item.product_id] = item.quantity;
        }
        return map;
      } catch {
        return {};
      }
    }
    ```

    **ui.ts (Shopify) — Process line items in order create/update handlers:**

    In `POST /ui/orders` and `POST /ui/orders/:id`, extract line items from form body:
    ```typescript
    function extractLineItems(data: Record<string, string>): any[] | undefined {
      const lineItems: any[] = [];
      for (const key of Object.keys(data)) {
        if (key.startsWith('line_product_')) {
          const productId = parseInt(data[key]);
          const qtyKey = `line_qty_${productId}`;
          const quantity = parseInt(data[qtyKey]) || 1;
          const product = fastify.stateManager.getProduct(productId);
          lineItems.push({
            product_id: productId,
            title: product?.title || '',
            price: product?.price || '0.00',
            quantity,
          });
        }
      }
      return lineItems.length > 0 ? lineItems : undefined;
    }
    ```

    Pass `line_items: extractLineItems(data)` to both createOrder and updateOrder calls. Also calculate `total_price` from line items if not explicitly set:
    ```typescript
    const lineItems = extractLineItems(data);
    let totalPrice = data.total_price;
    if (lineItems && lineItems.length > 0 && !totalPrice) {
      totalPrice = lineItems.reduce((sum, item) => {
        return sum + (parseFloat(item.price) || 0) * item.quantity;
      }, 0).toFixed(2);
    }
    ```
