        exp: Math.floor(Date.now()/1000)+3600, nbf: Math.floor(Date.now()/1000)-5,
        iat: Math.floor(Date.now()/1000), jti: 'jti-bad', sid: 'sid-bad',
      }).setProtectedHeader({ alg: 'HS256' }).sign(wrongKey);
