```typescript
interface AwsEventV1 {
  body: string | null;
  headers: Record<string, string | undefined>;
  isBase64Encoded: boolean;
  pathParameters: Record<string, string | undefined> | null;
  queryStringParameters: Record<string, string | undefined> | null;
  requestContext: any;
  stageVariables: Record<string, string | undefined> | null;
  httpMethod: string;
  multiValueHeaders: Record<string, string[] | undefined>;
  multiValueQueryStringParameters: Record<string, string[] | undefined>;
  path: string;
  resource: string;
}

// Response shape:
// { statusCode: number, headers: Record<string,string>, body: string }
// url_verification: { statusCode: 200, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ challenge }) }
// invalid sig: { statusCode: 401, body: '' }
```

HMAC signing (identical scheme to HTTPReceiver in Phase 19):
```typescript
import { createHmac } from 'node:crypto';
const SIGNING_SECRET = 'test-signing-secret-slck12-lambda';

function makeAwsEvent(body: string, secret = SIGNING_SECRET): AwsEventV1 {
  const ts = Math.floor(Date.now() / 1000);
  const sig = `v0=${createHmac('sha256', secret).update(`v0:${ts}:${body}`).digest('hex')}`;
  return {
    body,
    headers: {
      'X-Slack-Signature': sig,
      'X-Slack-Request-Timestamp': String(ts),
      'Content-Type': 'application/json',
    },
    isBase64Encoded: false,
    httpMethod: 'POST',
    path: '/slack/events',
    resource: '/slack/events',
    pathParameters: null,
    queryStringParameters: null,
    multiValueHeaders: {},
    multiValueQueryStringParameters: {},
    requestContext: {} as any,
    stageVariables: null,
  };
}
```

Usage pattern (from research):
```typescript
import { AwsLambdaReceiver, App } from '@slack/bolt';

// URL verification — no app needed
const receiver = new AwsLambdaReceiver({ signingSecret: SIGNING_SECRET });
const handler = await receiver.start();  // returns the handler function directly
const response = await handler(makeAwsEvent(body), {} as any, () => {});

// Event delivery — app with listener needed
const app = new App({ receiver, token: TOKEN, clientOptions: { slackApiUrl: ... } });
app.event('app_mention', async ({ ack }) => { await ack(); listenerFired = true; });
// NOTE: receiver.start() already called above — App uses the already-started receiver
```

Critical: unhandledRequestTimeoutMillis (from research pitfalls):
- Default is 3001ms — causes spurious console.error after 3s if ack timing edge cases
- Set `unhandledRequestTimeoutMillis: 100` to keep tests fast and avoid log noise

TOKEN seeding: Use seedSlackBotToken for the bot token used by App's WebClient.
Use unique prefix 'xoxb-slck12-lambda-' to avoid conflicts with slck12 socket mode test.

```typescript
/**
 * SLCK-12: AwsLambdaReceiver conformance test
 *
 * Pure in-process harness — no network, no HTTP server, no AWS SDK required.

import { describe, it, expect, beforeAll } from 'vitest';
import { AwsLambdaReceiver, App } from '@slack/bolt';
import { createHmac } from 'node:crypto';
import { resetSlack, seedSlackBotToken } from '../setup/seeders.js';

const SIGNING_SECRET = 'test-signing-secret-slck12-lambda';
let TOKEN: string;

beforeAll(async () => {
  await resetSlack();
  TOKEN = await seedSlackBotToken('xoxb-slck12-lambda-bot-token');
});
```

**makeAwsEvent helper** (shown in interfaces — use exactly that shape):
Build AwsEventV1 with correct HMAC headers. Add an overload for invalid-sig test: pass a wrong secret.

**Test 1 — url_verification returns challenge:**
```typescript
it('returns 200 with challenge for url_verification', async () => {
  const receiver = new AwsLambdaReceiver({
    signingSecret: SIGNING_SECRET,
    unhandledRequestTimeoutMillis: 100,
  });
  const handler = await receiver.start();
  const body = JSON.stringify({ type: 'url_verification', challenge: 'test-challenge-slck12' });
  const response = await handler(makeAwsEvent(body), {} as any, () => {}) as any;
  expect(response.statusCode).toBe(200);
  const parsed = JSON.parse(response.body);
  expect(parsed.challenge).toBe('test-challenge-slck12');
});
```

**Test 2 — invalid HMAC returns 401:**
```typescript
it('returns 401 for invalid HMAC signature', async () => {
  const receiver = new AwsLambdaReceiver({
    signingSecret: SIGNING_SECRET,
    unhandledRequestTimeoutMillis: 100,
  });
  const handler = await receiver.start();
  const body = JSON.stringify({ type: 'url_verification', challenge: 'test-challenge' });
  // Sign with a DIFFERENT secret to produce invalid sig

});
```

**Test 3 — event delivery fires listener:**
```typescript
it('delivers event_callback payload to app.event() listener', async () => {
  const receiver = new AwsLambdaReceiver({
    signingSecret: SIGNING_SECRET,
    unhandledRequestTimeoutMillis: 100,
  });
  const app = new App({
    receiver,
    token: TOKEN,
    clientOptions: { slackApiUrl: process.env.SLACK_API_URL! + '/api/' },
  });

  let listenerFired = false;
  app.event('app_mention', async ({ ack }) => {
    await ack();
    listenerFired = true;

  const handler = await receiver.start();

    type: 'event_callback',
    team_id: 'T_TWIN',
    api_app_id: 'A_TWIN',
    event: {
      type: 'app_mention',
      user: 'U_TEST',
      text: '<@U_BOT_TWIN> hello lambda',
      channel: 'C_GENERAL',
      event_ts: '1234567890.000200',
    },
    event_id: 'Ev_SLCK12_LAMBDA',
    event_time: 1234567890,
    authorizations: [{
      enterprise_id: null,
      team_id: 'T_TWIN',
      user_id: 'U_BOT_TWIN',
      is_bot: true,
      is_enterprise_install: false,
    }],
  });

  const response = await handler(makeAwsEvent(eventBody), {} as any, () => {}) as any;
  expect(response.statusCode).toBe(200);
  expect(listenerFired).toBe(true);
});
```
