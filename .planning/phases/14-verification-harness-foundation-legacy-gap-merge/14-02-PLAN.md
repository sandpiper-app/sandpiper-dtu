    **tests/sdk-verification/helpers/shopify-client.ts:**
    ```typescript
    import { createAdminApiClient } from '@shopify/admin-api-client';

    /**
     * Create a Shopify admin-api-client wired to the local twin.
     *
     * CRITICAL: The SDK forces https:// on the storeDomain via validateDomainAndGetStoreUrl.
     * customFetchApi receives an https:// URL and must rewrite protocol+host+port to the
     * local HTTP twin. Never pass the twin's http:// URL as storeDomain.
     */
    export function createShopifyClient(options?: { apiVersion?: string }) {
      const twinBaseUrl = process.env.SHOPIFY_API_URL!;
      const twinUrl = new URL(twinBaseUrl);

      const customFetchApi: typeof fetch = async (input, init) => {
        const url = new URL(typeof input === 'string' ? input : input.toString());
        url.protocol = twinUrl.protocol;   // 'http:'
        url.hostname = twinUrl.hostname;   // '127.0.0.1'
        url.port = twinUrl.port;           // dynamic port string
        return fetch(url.toString(), init);
      };

      return createAdminApiClient({
        storeDomain: 'dev.myshopify.com',
        apiVersion: options?.apiVersion ?? '2025-07',
        accessToken: 'test-access-token',
        customFetchApi,
        isTesting: true,
      });
    }
    ```

    **tests/sdk-verification/helpers/slack-client.ts:**
    ```typescript
    import { WebClient } from '@slack/web-api';

    /**
     * Create a Slack WebClient wired to the local twin.
     *
     * slackApiUrl must end with '/' — WebClient appends the method name directly
     * (e.g., slackApiUrl + 'auth.test'). Omitting the trailing slash produces
     * malformed URLs like '/api/apiauth.test'.
     */
    export function createSlackClient(token?: string) {
      const slackApiUrl = process.env.SLACK_API_URL!;
      return new WebClient(token ?? 'xoxb-test-token', {
        slackApiUrl: slackApiUrl.replace(/\/$/, '') + '/api/',
      });
    }
    ```
