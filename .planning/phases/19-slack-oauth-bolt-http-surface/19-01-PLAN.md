    Key patterns from research:
    - Use `stateVerification: false` to skip cookie round-trip — simplest approach, fully covers SLCK-09
    - Pass `clientOptions: { slackApiUrl: process.env.SLACK_API_URL! + '/api/' }` to every InstallProvider instance — without this, internal WebClient calls go to real Slack API
    - Pass `authorizationUrl: process.env.SLACK_API_URL! + '/oauth/v2/authorize'` to redirect the install URL to the twin

    - Simulate `IncomingMessage` + `ServerResponse` from `node:http` using Socket from `node:net`

    ```typescript
    import { describe, it, expect, beforeAll } from 'vitest';
    import { InstallProvider, MemoryInstallationStore } from '@slack/oauth';
    import { IncomingMessage, ServerResponse } from 'node:http';
    import { Socket } from 'node:net';
    import { resetSlack } from '../setup/seeders.js';

    function makeInstaller(installationStore: MemoryInstallationStore) {
      const slackApiUrl = process.env.SLACK_API_URL! + '/api/';
      return new InstallProvider({
        clientId: 'test-client-id-19',
        clientSecret: 'test-client-secret-19',
        stateVerification: false,
        authorizationUrl: process.env.SLACK_API_URL! + '/oauth/v2/authorize',
        clientOptions: { slackApiUrl },
        installationStore,
      });
    }

    function makeReq(urlPath: string, headers: Record<string, string> = {}): IncomingMessage {
      const req = new IncomingMessage(new Socket());
      req.url = urlPath;
      Object.assign(req.headers, headers);
      return req;
    }

    function makeRes() {
      const captured = { headers: {} as Record<string, string | string[]>, statusCode: 200, body: '' };
      const res = {
        setHeader(k: string, v: string | string[]) { captured.headers[k] = v; },
        getHeader(k: string) { return captured.headers[k]; },
        writeHead(code: number) { captured.statusCode = code; },
        end(b = '') { captured.body = String(b); },
      } as unknown as ServerResponse;
      return { res, captured };
    }
    ```
