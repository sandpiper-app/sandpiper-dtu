HMAC signing helper (from bolt-js verify-request.ts — exact format required):
```typescript
import { createHmac } from 'node:crypto';
import type { AddressInfo } from 'node:net';

// JSON payloads (events, url_verification)
function signJsonRequest(body: string, signingSecret: string): Record<string, string> {
  const ts = Math.floor(Date.now() / 1000);
  const sigBase = `v0:${ts}:${body}`;
  const sig = `v0=${createHmac('sha256', signingSecret).update(sigBase).digest('hex')}`;
  return {
    'x-slack-signature': sig,
    'x-slack-request-timestamp': String(ts),
    'content-type': 'application/json',
  };
}

// Form-encoded payloads (slash commands, interactive payloads)
// CRITICAL: Bolt verifies HMAC over the raw form body, not JSON
function signFormRequest(body: string, signingSecret: string): Record<string, string> {
  const ts = Math.floor(Date.now() / 1000);
  const sigBase = `v0:${ts}:${body}`;
  const sig = `v0=${createHmac('sha256', signingSecret).update(sigBase).digest('hex')}`;
  return {
    'x-slack-signature': sig,
    'x-slack-request-timestamp': String(ts),
    'content-type': 'application/x-www-form-urlencoded',
  };
}
```

    ```typescript
    import { App, HTTPReceiver, ExpressReceiver } from '@slack/bolt';
    import { createHmac } from 'node:crypto';
    import { createServer } from 'node:http';
    import type { AddressInfo } from 'node:net';
    import { resetSlack, seedSlackBotToken } from '../setup/seeders.js';

    const SIGNING_SECRET = 'test-signing-secret-slck11';

    function signRequest(body: string, secret = SIGNING_SECRET, contentType = 'application/json'): Record<string, string> {
      const ts = Math.floor(Date.now() / 1000);
      const sig = `v0=${createHmac('sha256', secret).update(`v0:${ts}:${body}`).digest('hex')}`;
      return {
        'x-slack-signature': sig,
        'x-slack-request-timestamp': String(ts),
        'content-type': contentType,
      };
    }
    ```

    ```typescript
    app.event('app_mention', async ({ event }) => { /* capture */ });
    const server = createServer(receiver.app);
    await new Promise<void>(resolve => server.listen(0, '127.0.0.1', resolve));
    const port = (server.address() as AddressInfo).port;
    try {
      // 5a: url_verification
      const uvBody = JSON.stringify({ type: 'url_verification', challenge: 'express-challenge' });
      const uvRes = await fetch(`http://127.0.0.1:${port}/slack/events`, {
        method: 'POST', headers: signRequest(uvBody), body: uvBody,

      const badRes = await fetch(`http://127.0.0.1:${port}/slack/events`, {
    ```

    **Test 6 — respond() in command handler:**
    Register `app.command('/respond-test', async ({ ack, respond }) => { await ack(); await respond('pong'); })`.
    CRITICAL: Slash commands use `application/x-www-form-urlencoded` content type, not JSON. Build the body as a URL-encoded form string:
    ```typescript
    const formBody = new URLSearchParams({
      command: '/respond-test', text: 'hello', team_id: 'T_TWIN',
      user_id: 'U_TEST', channel_id: 'C_GENERAL', trigger_id: 'trig_19',
    ```

    Note on response_url: The twin's `/response-url/:id` may return 410 if `test-resp-19` was not pre-registered via an interaction trigger. The test should handle two scenarios:

    The primary goal is proving the command handler receives `respond` and invokes it correctly — the twin endpoint behavior is secondary.

    Anti-patterns:
