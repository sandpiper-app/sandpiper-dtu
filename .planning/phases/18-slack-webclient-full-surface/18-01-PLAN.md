**files.ts** — Create `twins/slack/src/plugins/web-api/files.ts`. This plugin serves three endpoints required by `filesUploadV2`:

```typescript
import type { FastifyPluginAsync } from 'fastify';
import { extractToken } from '../../services/token-validator.js';
import type { SlackStateManager } from '../../state/slack-state-manager.js';

declare module 'fastify' {
  interface FastifyInstance { slackStateManager: SlackStateManager; }
}

const filesPlugin: FastifyPluginAsync = async (fastify) => {
  // POST /api/files.getUploadURLExternal — step 1 of filesUploadV2 chain
  fastify.post('/api/files.getUploadURLExternal', async (request, reply) => {
    const token = extractToken(request);
    if (!token) return reply.send({ ok: false, error: 'not_authed' });
    const tokenRecord = fastify.slackStateManager.getToken(token);
    if (!tokenRecord) return reply.send({ ok: false, error: 'invalid_auth' });
    const { filename } = (request.body as any) ?? {};
    const file_id = `F_${Math.random().toString(36).substring(2, 11).toUpperCase()}`;
    // CRITICAL: upload_url MUST be absolute — SDK calls it directly, not via slackApiUrl
    // Read SLACK_API_URL per-request (set by globalSetup AFTER twin boots)
    const baseUrl = (process.env.SLACK_API_URL ?? '').replace(/\/api\/?$/, '');
    const upload_url = `${baseUrl}/api/_upload/${file_id}`;
    return { ok: true, file_id, upload_url };
  });

  // PUT /api/_upload/:file_id — step 2 of filesUploadV2 chain (binary upload)
  // No auth required — SDK calls upload_url directly using axios, not through WebClient auth
  fastify.put('/api/_upload/:file_id', async () => {
    // Accept upload, return 200. No state storage needed for conformance.
    return {};
  });

  // POST /api/files.completeUploadExternal — step 3 of filesUploadV2 chain
  fastify.post('/api/files.completeUploadExternal', async (request, reply) => {
    const token = extractToken(request);
    if (!token) return reply.send({ ok: false, error: 'not_authed' });
    const tokenRecord = fastify.slackStateManager.getToken(token);
    if (!tokenRecord) return reply.send({ ok: false, error: 'invalid_auth' });
    const { files } = (request.body as any) ?? {};
    const completed = (files ?? []).map((f: any) => ({ id: f.id, title: f.title ?? 'Uploaded file' }));
    return { ok: true, files: completed };
  });
};

export default filesPlugin;
export { filesPlugin };
```

**chat.ts expansion** — Add 11 new route handlers to the existing `chatPlugin`. Keep the existing postMessage and update handlers intact. The new handlers follow the same auth/rate/error-sim pattern. All use POST only (not GET+POST — these are write methods).

New handlers to add inside the `chatPlugin` async function:

1. `POST /api/chat.delete` — requires `ts` and `channel` in body; call `fastify.slackStateManager.deleteMessage(ts)` if it exists (or skip silently — twin is lenient); return `{ ok: true, channel, ts }`.

2. `POST /api/chat.postEphemeral` — requires `channel` and `user`; return `{ ok: true, message_ts: generateMessageTs() }`. No state storage.

3. `GET+POST /api/chat.getPermalink` — requires `channel` and `message_ts`; return `{ ok: true, channel, permalink: 'https://twin-workspace.slack.com/archives/' + channel + '/p' + (message_ts ?? '').replace('.', '') }`.

4. `POST /api/chat.meMessage` — requires `channel` and `text`; create message with ts via `generateMessageTs()`, store via `fastify.slackStateManager.createMessage(...)`, return `{ ok: true, channel, ts }`.

5. `POST /api/chat.scheduleMessage` — requires `channel` and `post_at`; return `{ ok: true, channel, scheduled_message_id: 'SM' + Date.now(), post_at: Number(post_at) }`. No state storage.

6. `GET+POST /api/chat.scheduledMessages.list` — return `{ ok: true, scheduled_messages: [], response_metadata: { next_cursor: '' } }`.

7. `POST /api/chat.deleteScheduledMessage` — return `{ ok: true }`.

8. `POST /api/chat.unfurl` — return `{ ok: true }`.

9. `POST /api/chat.startStream` — requires `channel`; generate ts; return `{ ok: true, ts }`. CRITICAL: `ts` field name in response is what ChatStreamer reads to track the stream.

10. `POST /api/chat.appendStream` — requires `ts`; return `{ ok: true }`.

11. `POST /api/chat.stopStream` — requires `ts`; return `{ ok: true }`.

All new handlers need the same auth/rate/error-sim preamble. For chat.delete, add `deleteMessage` to the slackStateManager call if the method exists; if not, do a silent no-op (check with `if (fastify.slackStateManager.getMessage(ts)) { ... }`).

    const channel = await seedSlackChannel('uploads');
    const client = createSlackClient(token);
    await expect(
      client.filesUploadV2({ filename: 'test.txt', data: Buffer.from('hello'), channel_id: channel })

  it('ChatStreamer append/stop sequence completes', async () => {
    const channel = await seedSlackChannel('streams');
    const client = createSlackClient(token);
    const streamer = await client.chat.startStream({ channel });
    // startStream returns a ChatStreamer instance
    expect(streamer).toBeDefined();
    if (streamer instanceof ChatStreamer) {
      await streamer.append({ text: 'chunk one' });
      await streamer.stop({ text: 'final' });
    }
  });

  it('twin returns ratelimited error when error simulator is set', async () => {
    const slackApiUrl = process.env.SLACK_API_URL!;
    // Configure error simulator to return 429 for auth.test
    await fetch(`${slackApiUrl}/admin/errors`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ operation: 'auth.test', status_code: 429, error_body: JSON.stringify({ ok: false, error: 'ratelimited' }) }),
    });

    // Client with rejectRateLimitedCalls: true — throws on 429
    const client = createSlackClient(token);
    // The raw call should get a 429 back from the twin
    const result = await fetch(`${slackApiUrl}/api/auth.test`, {
      method: 'POST',
      headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
      body: '{}',
    });
    expect(result.status).toBe(429);

    // Clear error config — normal call succeeds
    await fetch(`${slackApiUrl}/admin/errors`, {
      method: 'DELETE',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ operation: 'auth.test' }),
    });
    const ok = await client.auth.test();
    expect(ok.ok).toBe(true);
  });
});
