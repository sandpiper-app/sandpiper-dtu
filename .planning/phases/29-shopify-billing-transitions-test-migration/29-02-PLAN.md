Established token seeder pattern (from billing-state-machine.test.ts):
```typescript
import { randomUUID } from 'node:crypto';

async function seedToken(
  app: Awaited<ReturnType<typeof buildApp>>,
  shopDomain: string
): Promise<string> {
  const token = randomUUID();
  await app.inject({
    method: 'POST',
    url: '/admin/tokens',
    payload: { token, shopDomain },
  });
  return token;
}
```
